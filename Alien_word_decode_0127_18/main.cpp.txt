char Answer[6];
char Input[6];
char Output[6];
int M = 32;

char Word[20000][6];
char Phone[32];

int IsSame(char a[6], char b[6])
{
	int ret = 1;
	for (int i = 0; i < 6; i++) {
		if (a[i] != b[i]) {
			ret = 0;
			break;
		}
	}
	return ret;
}

int main()
{
	freopen("UserInput.txt", "rt", stdin);

		for (i = 0; i < 6; i++) {

		}

				scanf("%02x", &Word[j][i]);
			}
		}

		learn_phone(32,Phone); // 1 ¡Â M ¡Â 32
		learn_word(20000, Word); // 1 ¡Â N ¡Â 20000
		recognize(Input, Output); // restore is called by 20,000. in, out: char[6]
